快速开始

# 安装 Agent Framework
pip install agent-framework

# 创建你的第一个智能体
from agent_framework import Agent

agent = Agent(
    name="MyAgent",
    capabilities=["reasoning", "learning", "acting"]
)

# 启动智能体
agent.start()
agent.execute_task("完成某个任务")
agent.stop()
